前言

本文整理并总结了十大经典的排序算法（冒泡排序、选择排序、插入排序、快速排序、归并排序、希尔排序、计数排序、基数排序、桶排序、堆排序）的时间复杂度、空间复杂度等性质。

本文并不会详细讲解每种排序算法的原理，网上有很多很好的教程，大家可以自己去搜了看。

最后我还亲自手写了十种排序算法的 c++ 代码，大家可以用来通过 LeetCode 912. 排序数组 这道题。

代码实现

所有的排序算法接口都是相同的，也就是 vector<int> xxxSort(vector<int>& nums) 。只需要你传入一个 vector<int> 类型的数组，就能返回排序后的结果。

运行下来可以发现，桶排序速度是比较快的。而冒泡排序、选择排序和插入排序因为时间复杂度太高无法通过本题，基数排序因为无法处理负数也不能通过本题。

class Solution &#123;
public:
    vector<int> sortArray(vector<int>& nums) &#123;
        return quickSort(nums);
    &#125;

    // 冒泡排序（超时）
    vector<int> bubbleSort(vector<int>& nums) &#123;
        int n = nums.size();
        for (int i = 0; i < n; ++i) &#123;
            for (int j = n-2; j >= i; --j) &#123;
                if (nums[j] > nums[j+1]) &#123;
                    swap(nums[j], nums[j+1]);
                &#125;
            &#125;
        &#125;
        return nums;
    &#125;

    // 选择排序（超时）
    vector<int> selectSort(vector<int>& nums) &#123;
        int n = nums.size();
        for (int i = 0; i < n; ++i) &#123;
            int idx = i;
            for (int j = i; j < n; ++j) &#123;
                if (nums[j] < nums[idx]) &#123;
                    idx = j;
                &#125;
            &#125;
            swap(nums[i], nums[idx]);
        &#125;
        return nums;
    &#125;

    // 插入排序（超时）
    vector<int> insertSort(vector<int>& nums) &#123;
        int n = nums.size();
        for (int i = 0; i < n; ++i) &#123;
            for (int j = i; j > 0 && nums[j] < nums[j-1]; --j) &#123;
                swap(nums[j], nums[j-1]);
            &#125;
        &#125;
        return nums;
    &#125;

    // 快速排序（24 ms）
    void qSort(vector<int>& nums, int l, int r) &#123;
        if (l >= r) return;
        int m = l;
        for (int i = l; i < r; ++i) &#123;
            if (nums[i] < nums[r]) &#123;
                swap(nums[m++], nums[i]);
            &#125;
        &#125;
        swap(nums[m], nums[r]);
        qSort(nums, l, m-1);
        qSort(nums, m+1, r);
    &#125;

    vector<int> quickSort(vector<int>& nums) &#123;
        int n = nums.size();
        qSort(nums, 0, n-1);
        return nums;
    &#125;

    // 归并排序（192 ms）
    vector<int> mSort(vector<int>& nums, int l, int r) &#123;
        if (l >= r) return &#123;nums[l]&#125;;
        int m = l+(r-l)/2;
        vector<int> lnums = mSort(nums, l, m);
        vector<int> rnums = mSort(nums, m+1, r);
        vector<int> res;
        int i = 0, j = 0;
        while (i <= m-l && j <= r-m-1) &#123;
            if (lnums[i] < rnums[j]) &#123;
                res.push_back(lnums[i++]);
            &#125; else &#123;
                res.push_back(rnums[j++]);
            &#125;
        &#125;
        while (i <= m-l) &#123;
            res.push_back(lnums[i++]);
        &#125;
        while (j <= r-m-1) &#123;
            res.push_back(rnums[j++]);
        &#125;
        return res;
    &#125;

    vector<int> mergeSort(vector<int>& nums) &#123;
        int n = nums.size();
        nums = mSort(nums, 0, n-1);
        return nums;
    &#125;

    // 归并排序 + 非递归（80 ms）
    vector<int> mergeSortNR(vector<int>& nums) &#123;
        int n = nums.size();
        for (int len = 1; len < n; len <<= 1) &#123;
            for (int l = 0; l < n-len; l += 2*len) &#123;
                int m = l+len-1;
                int r = min(n-1, l+2*len-1);
                vector<int> res;
                int i = l, j = m+1;
                while (i <= m && j <= r) &#123;
                    if (nums[i] < nums[j]) &#123;
                        res.push_back(nums[i++]);
                    &#125; else &#123;
                        res.push_back(nums[j++]);
                    &#125;
                &#125;
                while (i <= m) &#123;
                    res.push_back(nums[i++]);
                &#125;
                while (j <= r) &#123;
                    res.push_back(nums[j++]);
                &#125;
                for (int i = l; i <= r; ++i) &#123;
                    nums[i] = res[i-l];
                &#125;
            &#125;
        &#125;
        return nums;
    &#125;

    // 希尔排序（40 ms）
    vector<int> shellSort(vector<int>& nums) &#123;
        int n = nums.size();
        for (int gap = n/2; gap > 0; gap /= 2) &#123;
            for (int i = gap; i < n; ++i) &#123;
                for (int j = i; j-gap >= 0 && nums[j-gap] > nums[j]; j -= gap) &#123;
                    swap(nums[j-gap], nums[j]);
                &#125;
            &#125;
        &#125;
        return nums;
    &#125;

    // 计数排序（32 ms）
    vector<int> countSort(vector<int>& nums) &#123;
        int n = nums.size();
        if (!n) return &#123;&#125;;
        int minv = *min_element(nums.begin(), nums.end());
        int maxv = *max_element(nums.begin(), nums.end());
        int m = maxv-minv+1;
        vector<int> count(m, 0);
        for (int i = 0; i < n; ++i) &#123;
            count[nums[i]-minv]++;
        &#125;
        vector<int> res;
        for (int i = 0; i < m; ++i) &#123;
            for (int j = 0; j < count[i]; ++j) &#123;
                res.push_back(i+minv);
            &#125;
        &#125;
        return res;
    &#125;

    // 基数排序（不适用于负数）
    vector<int> radixSort(vector<int>& nums) &#123;
        int n = nums.size();
        int maxv = *max_element(nums.begin(), nums.end());
        int maxd = 0;
        while (maxv > 0) &#123;
            maxv /= 10;
            maxd++;
        &#125;
        vector<int> count(10, 0), rank(n, 0);
        int base = 1;
        while (maxd > 0) &#123;
            count.assign(10, 0);
            for (int i = 0; i < n; ++i) &#123;
                count[(nums[i]/base)%10]++;
            &#125;
            for (int i = 1; i < 10; ++i) &#123;
                count[i] += count[i-1];
            &#125;
            for (int i = n-1; i >= 0; --i) &#123;
                rank[--count[(nums[i]/base)%10]] = nums[i];
            &#125;
            for (int i = 0; i < n; ++i) &#123;
                nums[i] = rank[i];
            &#125;
            maxd--;
            base *= 10;
        &#125;
        return nums;
    &#125;

    // 桶排序 (20 ms)
    vector<int> bucketSort(vector<int>& nums) &#123;
        int n = nums.size();
        int maxv = *max_element(nums.begin(), nums.end());
        int minv = *min_element(nums.begin(), nums.end());
        int bs = 1000;
        int m = (maxv-minv)/bs+1;
        vector<vector<int> > bucket(m);
        for (int i = 0; i < n; ++i) &#123;
            bucket[(nums[i]-minv)/bs].push_back(nums[i]);
        &#125;
        int idx = 0;
        for (int i = 0; i < m; ++i) &#123;
            int sz = bucket[i].size();
            bucket[i] = quickSort(bucket[i]);
            for (int j = 0; j < sz; ++j) &#123;
                nums[idx++] = bucket[i][j];
            &#125;
        &#125;
        return nums;
    &#125;

    // 堆排序（32 ms）
    void adjust(vector<int>& nums, int p, int s) &#123;
        while (2*p+1 < s) &#123;
            int c1 = 2*p+1;
            int c2 = 2*p+2;
            int c = (c2<s && nums[c2]>nums[c1]) ? c2 : c1;
            if (nums[c] > nums[p]) swap(nums[c], nums[p]);
            else break;
            p = c;
        &#125;
    &#125;

    vector<int> heapSort(vector<int>& nums) &#123;
        int n = nums.size();
        for (int i = n/2-1; i >= 0; --i) &#123;
            adjust(nums, i, n);
        &#125;
        for (int i = n-1; i > 0; --i) &#123;
            swap(nums[0], nums[i]);
            adjust(nums, 0, i);
        &#125;
        return nums;
    &#125;
